Переписать через стримы фрагменты кода:
for (String s : strings) {
System.out.println(s);
}


public static List method(List strings) {
    List result = new ArrayList<>();
    for (String s : strings) {
        if (!result.contains(s)) {
            result.add(s);
        }
    }
    return result;
}
